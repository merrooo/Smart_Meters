/**
 * نظام معالجة بيانات العدادات المركزي - الإصدار المعتمد
 * المرجع الأساسي: رقم العداد
 */

let finalDataMap = new Map();
let categorizedData = { A1: [], A2: [], A3: [], A4: [], A5: [], ALL: [] };

// الهيكل النهائي الصارم (41 عموداً)
const FULL_ORIGINAL_HEADER = [
    "كود المشترك", "فرع", "منطقة", "يومية", "مرجع", "فرعى", "وحده", "اسم المشترك", 
    "الرقم القومي", "العنوان", "رقم التليفون", "نوع الرسائل", "نوع الرسائل2", 
    "كود السداد", "حالة الاشتراك", "الجهد", "القدرة", "معامل الضرب", "رقم العداد", 
    "SIM_S_Num", "BadgeNumber", "Manufacturer", "Model", "Name of connected Feeder", 
    "Name of connected KIOSK", "Disco", "Address (General Region)", "Address (Region)", 
    "Area", "LossesArea", "DeviceFunction", "DistributionFacility", "تاريخ التعاقد", 
    "نوع التعاقد// العداد", "النشاط", "Address4", "TYPE", "LatitudeY", "LongitudeX", 
    "DivisionCode", "Date"
];

// مرادفات الحقول لضمان جلب البيانات حتى لو اختلف المسمى في الملف المرفوع
const COLUMN_ALIASES = {
    meterNumber: ['رقم العداد', 'BadgeNumber', 'Meter_Number', 'Serial', 'Meter No', 'رقم_العداد'],
    multiplier: ['معامل الضرب', 'معامل_الضرب', 'Multiplier', 'Factor', 'معامل ضرب'],
    power: ['القدرة', 'القدره', 'Power', 'Capacity', 'القدرة_المعاقد_عليها']
};

/** --- دوال المساعدة --- **/

function findValueByAlias(row, aliasArray) {
    for (let alias of aliasArray) {
        let key = Object.keys(row).find(k => k.trim().toLowerCase() === alias.toLowerCase());
        if (key && row[key] !== undefined && row[key] !== null && String(row[key]).trim() !== "") {
            return row[key];
        }
    }
    return "";
}

async function readExcel(file) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = (e) => {
            try {
                const data = new Uint8Array(e.target.result);
                const workbook = XLSX.read(data, { type: 'array', cellDates: true });
                const firstSheetName = workbook.SheetNames[0];
                const jsonData = XLSX.utils.sheet_to_json(workbook.Sheets[firstSheetName], { defval: "" });
                resolve(jsonData);
            } catch (err) { reject(err); }
        };
        reader.readAsArrayBuffer(file);
    });
}

/** --- المعالجة والدمج --- **/
/** --- المعالجة والدمج المحسنة --- **/

async function processAllFiles(files) {
    finalDataMap.clear();
    
    for (const file of files) {
        const jsonRows = await readExcel(file);
        
        jsonRows.forEach(row => {
            // استخراج رقم العداد (المرجع الأساسي)
            const meterNo = findValueByAlias(row, COLUMN_ALIASES.meterNumber).toString().trim();
            if (!meterNo) return;

            let existing = finalDataMap.get(meterNo) || {};
            let merged = { ...existing };

            // ملء الأعمدة الـ 41 عموداً بعمود
            FULL_ORIGINAL_HEADER.forEach(header => {
                let val = "";
                
                if (header === "رقم العداد") {
                    val = meterNo;
                } else if (header === "معامل الضرب") {
                    // البحث عن القيمة في الملفات المرفوعة
                    let foundVal = findValueByAlias(row, COLUMN_ALIASES.multiplier);
                    // إذا كانت فارغة تماماً، نجعلها "-" ولا نضع "1"
                    val = (foundVal !== undefined && foundVal !== null && foundVal !== "") ? foundVal : "-";
                } else if (header === "القدرة") {
                    let foundPower = findValueByAlias(row, COLUMN_ALIASES.power);
                    val = (foundPower !== undefined && foundPower !== null && foundPower !== "") ? foundPower : "-";
                } else {
                    val = row[header] !== undefined ? row[header] : "";
                }

                // تحديث البيانات في السجل المدمج
                // نحدث القيمة إذا كان السجل الحالي فارغاً أو يحتوي على "-" بينما الملف الجديد يحتوي على قيمة فعلية
                if ((merged[header] === undefined || merged[header] === "" || merged[header] === "-") && (val !== "" && val !== "-")) {
                    merged[header] = val;
                } 
                // إذا لم تكن هناك قيمة قديمة أصلاً، نضع القيمة الحالية (حتى لو كانت "-")
                else if (merged[header] === undefined) {
                    merged[header] = val;
                }
            });

            finalDataMap.set(meterNo, merged);
        });
    }
}

function categorizeMeters() {
    categorizedData = { A1: [], A2: [], A3: [], A4: [], A5: [], ALL: [] };
    
    finalDataMap.forEach((data) => {
        // تحديد النوع بناءً على عمود TYPE أو "نوع التعاقد// العداد"
        let typeRaw = (data["TYPE"] || data["نوع التعاقد// العداد"] || "").toString().toUpperCase();
        
        let category = ""; 
        if (typeRaw.includes('A1')) category = 'A1';
        else if (typeRaw.includes('A2')) category = 'A2';
        else if (typeRaw.includes('A3')) category = 'A3';
        else if (typeRaw.includes('A4')) category = 'A4';
        else if (typeRaw.includes('A5')) category = 'A5';

        if (category) categorizedData[category].push(data);
        categorizedData.ALL.push(data);
    });
}

/** --- التصدير والتحميل --- **/

function exportToExcel(category) {
    const rawData = categorizedData[category];
    if (!rawData || rawData.length === 0) {
        Swal.fire('تنبيه', `لا توجد بيانات للفئة ${category}`, 'info');
        return;
    }

    // بناء المصفوفة لضمان الترتيب الصارم للأعمدة
    const worksheetData = rawData.map(item => {
        return FULL_ORIGINAL_HEADER.map(header => item[header] || "");
    });

    // إضافة الهيدر في أول صف
    worksheetData.unshift(FULL_ORIGINAL_HEADER);

    const ws = XLSX.utils.aoa_to_sheet(worksheetData);
    const wb = XLSX.utils.book_new();
    const fileName = category === 'ALL' ? "General_meters_all.xlsx" : `${category}_meters_data.xlsx`;
    
    XLSX.utils.book_append_sheet(wb, ws, "Meters_Data");
    XLSX.writeFile(wb, fileName);
}

/** --- أحداث الواجهة (Events) --- **/

document.addEventListener('DOMContentLoaded', () => {
    const fileInput = document.getElementById('fileInput');
    const processBtn = document.getElementById('processBtn');

    // عند اختيار الملفات
    fileInput.addEventListener('change', (e) => {
        const files = e.target.files;
        if (files.length > 0) {
            document.getElementById('fileListContainer').style.display = 'block';
            const list = document.getElementById('fileList');
            list.innerHTML = '';
            Array.from(files).forEach(f => {
                const li = document.createElement('li');
                li.textContent = `📄 ${f.name}`;
                list.appendChild(li);
            });
        }
    });

    // عند الضغط على زر المعالجة
    processBtn.addEventListener('click', async () => {
        if (fileInput.files.length === 0) return;

        Swal.fire({
            title: 'جاري معالجة البيانات...',
            text: 'يتم الآن فحص العدادات ودمج السجلات المكررة',
            allowOutsideClick: false,
            didOpen: () => Swal.showLoading()
        });

        try {
            await processAllFiles(fileInput.files);
            categorizeMeters();

            document.getElementById('uniqueCount').innerText = finalDataMap.size;
            document.getElementById('downloadSection').style.display = 'grid';

            Swal.fire('تمت العملية', `تم دمج ${finalDataMap.size} عداد فريد بنجاح`, 'success');
        } catch (error) {
            console.error(error);
            Swal.fire('خطأ', 'حدث خطأ أثناء معالجة الملفات', 'error');
        }
    });

    // ربط أزرار التحميل
    document.querySelectorAll('.btn-download, .btn-all').forEach(btn => {
        btn.addEventListener('click', function() {
            const type = this.getAttribute('data-type');
            exportToExcel(type);
        });
    });
});
